  "password": "password"
}</code></pre>
      </li>
      <li>
        <h4>Start the application:</h4>
        <code>npm start</code>
        or 
        <code>yarn start</code>
      </li>

        <h4>Launch cypress:</h4>
        <code>npm run cypress:open</code>
        or 
        <code>yarn run cypress:open</code>
